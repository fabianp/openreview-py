### Update super invitation or create a new one to perform bulk changes

{
    readers: ['.TMLR'],
    writers: ['.TMLR'],
    signatures '.TMLR',
    invitation: {
        id: '.TMLR/Review',
        edit: {
            invitation: {
                edit: {
                    note: {
                        readers: { const: { const: [ 'everyone' ] } }
                    }
                }
            }
        }
    }
}
